    def get_s3_client():
        params = {
            "region_name": settings.AWS_REGION,
            "config": Config(signature_version="s3v4")
        }
        if settings.S3_ENDPOINT:
            params["endpoint_url"] = settings.S3_ENDPOINT
        if settings.S3_ACCESS_KEY:
            params["aws_access_key_id"] = settings.S3_ACCESS_KEY
            params["aws_secret_access_key"] = settings.S3_SECRET_KEY
        return boto3.client("s3", **params)
